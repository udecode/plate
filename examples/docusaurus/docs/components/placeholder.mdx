### Installation

```bash npm2yarn
npm install @udecode/plate
# or
npm install @udecode/plate-ui-placeholder
```

### Usage

import { PlaceholderSandpack } from './PlaceholderSandpack'

<PlaceholderSandpack />

```ts live
() => {
  const components = withStyledPlaceHolders(createPlateUI());
  const plugins = createPlugins([
    ...basicNodesPlugins,
    createExitBreakPlugin(exitBreakPlugin),
  ], { components });

  return (
    <>
      <HeadingToolbar>
        <BasicElementToolbarButtons />
      </HeadingToolbar>
      <Plate
        id="placeholder"
        editableProps={editableProps}
        initialValue={placeholderValue}
        plugins={plugins}
      />
    </>
  );
}
```

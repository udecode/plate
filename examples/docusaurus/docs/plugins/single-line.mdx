Allows you to force the editor to only have one line.

### Installation

```bash npm2yarn
npm install @udecode/plate
# or
npm install @udecode/plate-break
```

### Usage

import { SingleLineSandpack } from './SingleLineSandpack'

<SingleLineSandpack />

```ts live
() => {
  const plugins = createPlugins([
    ...basicNodesPlugins,
    createListPlugin(),
    createTablePlugin(),
    createSingleLinePlugin(),
  ], {
    components: createPlateUI(),
  });

  return (
    <Plate
      id="single-line"
      editableProps={editableProps}
      initialValue={singleLineValue}
      plugins={plugins}
    />
  );
}
```

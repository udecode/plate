### Installation

```bash npm2yarn
npm install @udecode/plate-kbd
```

### Usage

```ts live
() => {
  const KbdToolbarButton = withPlateEventProvider(() => {
    const editor = usePlateEditorRef();

    return (
      <MarkToolbarButton
        type={getPluginType(editor, MARK_KBD)}
        icon={<Keyboard />}
      />
    );
  });

  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createKbdPlugin(),
  ], {
    components: CONFIG.components,
  });

  return (
    <>
      <HeadingToolbar>
        <KbdToolbarButton />
      </HeadingToolbar>
      <Plate
        id="kbd"
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.kbd}
        plugins={plugins}
      />
    </>
  );
}
```

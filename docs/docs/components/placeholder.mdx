### Installation

```bash npm2yarn
npm install @udecode/plate-ui-placeholder
```

### Usage

```ts live
() => {
  const components = withStyledPlaceHolders(CONFIG.components);
  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createExitBreakPlugin(CONFIG.exitBreak),
  ], { components });

  return (
    <>
      <HeadingToolbar>
        <BasicElementToolbarButtons />
      </HeadingToolbar>
      <Plate
        id="placeholder"
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.placeholder}
        plugins={plugins}
      />
    </>
  );
}
```

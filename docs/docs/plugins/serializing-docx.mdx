### Installation

```bash npm2yarn
npm install @udecode/plate-docx-serializer
```

### Deserializer Usage

```ts live
() => {
  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createImagePlugin(),
    createLinkPlugin(),
    createIndentListPlugin(),
    createIndentPlugin(),
    createTablePlugin(),
    createTodoListPlugin(),
    createMediaEmbedPlugin(),
    createHighlightPlugin(),
    createDeserializeDocxPlugin(),
  ], {
    components: CONFIG.components,
  });
    
  return (
    <Plate
      id="deserialize-docx"
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.pasteHtml}
      plugins={plugins}
    />
  );
}
```

```bash npm2yarn
npm install @udecode/plate-core
```

### Deserializer Usage

```ts live
() => {
  let plugins = [
    ...PLUGINS.basicNodes,
    createImagePlugin(),
    createLinkPlugin(),
    createListPlugin(),
    createTablePlugin(),
    createTodoListPlugin(),
    createMediaEmbedPlugin(),
    createExcalidrawPlugin(),
    createHighlightPlugin(),
    createMentionPlugin(),
    createSoftBreakPlugin(CONFIG.softBreak),
  ]

  plugins = createPlugins(plugins, {
    components: CONFIG.components,
  });

  return (
    <Plate
      id="deserialize-html"
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.pasteHtml}
      plugins={plugins}
    />
  );
}
```

### Serializer Usage

Use `serializeHtml`:

```ts live
() => {
  const editor = createPlateEditor({
    plugins: PLUGINS.basicNodes,
    components: CONFIG.components
  });

  const html = serializeHtml(editor, {
    nodes: VALUES.basicNodes,
  });

  return (
    <HighlightHTML code={html} />
  );
}
```

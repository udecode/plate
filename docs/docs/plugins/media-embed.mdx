Media providers supported by this plugin:

- video: 
  - youtube
  - vimeo
  - dailymotion
  - youku
  - coub
- twitter

### Installation

```bash npm2yarn
npm install @udecode/plate-media-embed
npm install @udecode/plate-ui-media-embed
```

### Usage

```ts live
() => {
  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createMediaEmbedPlugin(),
    createSelectOnBackspacePlugin({
      options: {
        query: {
          allow: [ELEMENT_MEDIA_EMBED],
        },
      },
    }),
  ], {
    components: CONFIG.components,
  });

  return (
    <>
      <PlateProvider id="media-embed">
        <HeadingToolbar>
          <MediaEmbedToolbarButton icon={<OndemandVideo />} />
        </HeadingToolbar>
      </PlateProvider>
   
      <Plate
        id="media-embed"
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.mediaEmbed}
        plugins={plugins}
      />
    </>
  );
}
```

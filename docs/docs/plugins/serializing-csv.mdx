### Installation

```bash npm2yarn
npm install @udecode/plate-csv-serializer
```

### Deserializer Usage

```ts live
() => {
  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createImagePlugin(),
    createLinkPlugin(),
    createListPlugin(),
    createTablePlugin(),
    createTodoListPlugin(),
    createMediaEmbedPlugin(),
    createExcalidrawPlugin(),
    createHighlightPlugin(),
    createMentionPlugin(),
    createSoftBreakPlugin(CONFIG.softBreak),
    createDeserializeCsvPlugin()
  ], {
    components: CONFIG.components,
  });

  return (
    <Plate
      id="deserialize-csv"
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.pasteCsv}
      plugins={plugins}
    />
  );
}
```

### Options

```tsx
export interface DeserializeCsvPlugin {
  /**
   * Percentage in decimal form, from 0 to a very large number, 0 for no errors allowed,
   * Percentage based on number of errors compared to number of rows
   * @default 0.25
   */
  errorTolerance?: number;
}
```

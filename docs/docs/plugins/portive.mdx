Add beautiful and intuitive images and attachments to Slate with support for paste, drag and drop and `<input type="file">` uploads.

See [Portive docs](https://slate-portive.docs.portive.com/).

### Installation

```bash npm2yarn
npm install @udecode/plate-portive
```

### Usage

```ts live
() => {
  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createImagePlugin({
      type: 'image-block'
    }),
    createPortivePlugin({
      apiOriginUrl: 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCIsImtpZCI6Imo0YlpVSGFQMmtlQmNyYTQifQ.eyJwYXRoIjoiKiovKiIsImlhdCI6MTY1NTU5OTg1MSwiZXhwIjoxNjg3MTU3NDUxfQ.WQpbtMbkH5sRvWgFNZrge1qLFb7FVNkdGzumbPPkm6M',
      path: "plate-portive",
    }),
  ], {
    components: {
      ...CONFIG.components,
      [ELEMENT_IMAGE]: PortiveImage,
    },
  });

  return (
    <>
      <PlateProvider id="image">
        <HeadingToolbar>
          <ImageToolbarButton icon={<Image />} />
        </HeadingToolbar>
      </PlateProvider>
      <Plate
        id="image"
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.image}
        plugins={plugins}
      />
    </>
  );
}
```

### Installation

```bash npm2yarn
npm install @udecode/plate-highlight
```

### Usage

```ts live
() => {
  const HighlightToolbarButton = withPlateEventProvider(() => {
    const editor = usePlateEditorRef();

    return (
      <MarkToolbarButton
        type={getPluginType(editor, MARK_HIGHLIGHT)}
        icon={<Highlight />}
      />
    );
  });

  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createHighlightPlugin(),
  ], {
    components: CONFIG.components,
  });

  return (
    <>
      <HeadingToolbar>
        <HighlightToolbarButton />
      </HeadingToolbar>
      <Plate
        id="highlight"
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.highlight}
        plugins={plugins}
      />
    </>
  );
}
```

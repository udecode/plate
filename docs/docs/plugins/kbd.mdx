### Installation

```bash npm2yarn
npm install @udecode/plate-kbd
```

### Usage

```ts live
() => {
  const KbdToolbarButton = () => {
    const editor = usePlateEditorRef();

    return (
      <MarkToolbarButton
        type={getPlatePluginType(editor, MARK_KBD)}
        icon={<Keyboard />}
      />
    );
  };

  const plugins = [
    ...PLUGINS.basicNodes,
    createKbdPlugin(),
  ];

  return (
    <>
      <HeadingToolbar>
        <KbdToolbarButton />
      </HeadingToolbar>
      <Plate
        id="kbd"
        plugins={plugins}
        components={CONFIG.components}
        options={CONFIG.options}
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.kbd}
      />
    </>
  );
}
```

## Usage

### Plugins

```tsx
import { createLinkPlugin } from '@udecode/plate-link';

const plugins = createPlugins(
  [
    // ...otherPlugins,
    createLinkPlugin(),
  ],
  {
    components: createPlateUI(),
  }
);
```

## Usage

### Plugins

```tsx
import { createTodoListPlugin } from '@udecode/plate-list';

const plugins = createPlugins(
  [
    // ...otherPlugins,
    createTodoListPlugin(),
  ],
  {
    components: createPlateUI(),
  }
);
```

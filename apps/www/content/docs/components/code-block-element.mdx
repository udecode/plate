## Usage

```tsx
import { createCodeBlockPlugin } from '@udecode/plate-code-block';
import { CodeBlockElement } from '@/components/plate-ui/code-Block-element';
```

```tsx
createPlugins([createCodeBlockPlugin()], {
  components: {
    [ELEMENT_CODE_LINE]: CodeBlockElement,
  },
});
```

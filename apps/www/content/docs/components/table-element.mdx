## Usage

### Plugins

```tsx
import { createTablePlugin } from '@udecode/plate-table';

const plugins = createPlugins(
  [
    // ...otherPlugins,
    createTablePlugin(),
  ],
  {
    components: createPlateUI(),
  }
);
```

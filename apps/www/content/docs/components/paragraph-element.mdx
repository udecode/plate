## Usage

### Plugins

```tsx
import { createParagraphPlugin } from '@udecode/plate-paragraph';

const plugins = createPlugins(
  [
    // ...otherPlugins,
    createParagraphPlugin(),
  ],
  {
    components: createPlateUI(),
  }
);
```
